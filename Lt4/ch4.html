5. Button《Label》

button = action + Label （本來是刪除 按了變成復原）

struct ContentView: View { @State var x="刪除" var body: some View { Button(action: {x="復原"}) {Label( title:{Text(x)}, icon:{Image(systemName:"trash.square") } ) } } }

button = action + Label （按了 刪除 ←→ 復原 ）

struct ContentView: View { @State var x:Bool = true var body: some View { Button(action:{x = !x}){ Label( title:{Text(x ? "刪除" : "復原")}, icon:{Image(systemName:"trash.square") } ) } } }

6. Button-buttonStyle-動畫

struct ContentView: View { @State var x:Bool = true var body: some View { VStack { Button(action:{x = !x}){ Image(systemName:"multiply") .resizable() .frame(width: 50.0, height: 50.0) }.buttonStyle(BlueButton()) } } } struct BlueButton: ButtonStyle { func makeBody(configuration: Configuration) -> some View { configuration.label .padding() .background(Color(red: 0.5, green: 0, blue: 0)) .foregroundColor(.white) .clipShape(Capsule()) .rotationEffect(.degrees(configuration.isPressed ? 180 :0 ) ) } }
